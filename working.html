Interesting Queries

Query 1 : Ask for all classes inside movies knowledge base

PREFIX ns0:     <http://dbpedia.org/property/>
PREFIX dbpprop: <http://dbpedia.org/property/>
PREFIX rdfs:    <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf:     <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT DISTINCT ?c WHERE 
{
    ?s rdf:type ?c .
} 
LIMIT 50

Query 4 : Ask for all films of a single actor

PREFIX ns4: <http://www.w3.org/ns/prov#>
PREFIX ns3: <http://dbpedia.org/datatype/>
PREFIX ns2: <http://dbpedia.org/ontology/>
PREFIX ns1: <http://dbpedia.org/ontology/Work/>
PREFIX ns0: <http://dbpedia.org/property/>
PREFIX dbo: <http://dbpedia.org/property/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dbpedia: <http://dbpedia.org/resource/>

SELECT DISTINCT ?film WHERE 
{
    ?film dbo:starring dbpedia:Akshaye_Khanna .
} 
LIMIT 50

Query 6 : Ask for all films where the producer name contains ‘Lee’ and the actor name contains ‘Sang-min’

PREFIX ns4: <http://www.w3.org/ns/prov#>
PREFIX ns3: <http://dbpedia.org/datatype/>
PREFIX ns2: <http://dbpedia.org/ontology/>
PREFIX ns1: <http://dbpedia.org/ontology/Work/>
PREFIX ns0: <http://dbpedia.org/property/>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dbo: <http://dbpedia.org/property/>
PREFIX dbpprop: <http://dbpedia.org/property/>

SELECT DISTINCT ?film WHERE 
{
    ?film dbpprop:producer ?producer .
    FILTER regex(str(?producer), "Lee")
           
        ?film dbo:starring ?actor .
    FILTER regex(str(?actor), "Sang-min")
} 
LIMIT 50
